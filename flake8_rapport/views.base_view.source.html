254 @staticmethod
255 def display_table(headers: List[str], rows: List[List[str]], title: str = ""):
256 """
257 Affiche un tableau formaté
258  
259 Args:
260 headers (List[str]): En-têtes des colonnes
261 rows (List[List[str]]): Lignes de données
262 title (str): Titre optionnel du tableau
263 """
264 if title:
265 BaseView.display_title(title)
266  
267 if not headers or not rows:
268 BaseView.display_info("Aucune donnée à afficher.")
269 return
270  
271 # Calculer les largeurs des colonnes
272 col_widths = [len(header) for header in headers]
273 for row in rows:
274 for i, cell in enumerate(row):
275 if i < len(col_widths):
276 col_widths[i] = max(col_widths[i], len(str(cell)))
277  
278 # Afficher les en-têtes
279 header_line = " | ".join(header.ljust(col_widths[i]) for i, header in enumerate(headers))
280 print(header_line)
281 BaseView.display_separator("-", len(header_line))
282  
283 # Afficher les lignes
284 for row in rows:
285 row_line = " | ".join(str(cell).ljust(col_widths[i]) for i, cell in enumerate(row))
286 print(row_line)

313 @staticmethod
314 def display_progress(current: int, total: int, message: str = "Progression"):
315 """
316 Affiche une barre de progression simple
317  
318 Args:
319 current (int): Valeur actuelle
320 total (int): Valeur totale
321 message (str): Message à afficher
322 """
323 if total <= 0:
324 return
325  
326 percentage = min(100, int((current / total) * 100))
327 bar_length = 30
328 filled_length = int((current / total) * bar_length)
329  
330 bar = "█" * filled_length + "░" * (bar_length - filled_length)
331 print(f"\r{message}: [{bar}] {percentage}% ({current}/{total})", end="", flush=True)
332  
333 if current >= total:
334 print() # Nouvelle ligne à la fin
